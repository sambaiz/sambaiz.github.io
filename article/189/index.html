MySQLのtime_zoneとgo-sql-driver/mysqlのlocの関係

2018-10-02 databasemysqlgolang

go-sql-driver/mysql で MySQL に接続するとき適切なパラメータを渡さないと日時が意図しない値になる。

> select version();
    +-----------+
    | version() |
    +-----------+
    | 5.7.21    |
    +-----------+

前提

タイムゾーンがDBにロードされていない場合はロードする。

> select count(*) from mysql.time_zone \\G;
*************************** 1. row ***************************
count(*): 0

$ mysql_tzinfo_to_sql /usr/share/zoneinfo/ | mysql -u root mysql

time_zone はデフォルト値の SYSTEM。つまり JST で、これは my.cnf の default-time-zone で変更できる。 関数 NOW() も JST の時間を返している。

> show variables like '%time_zone%';
+------------------+--------+
| Variable_name    | Value  |
+------------------+--------+
| system_time_zone | JST    |
| time_zone        | SYSTEM |
+------------------+--------+

> SELECT NOW();
mysql> SELECT NOW() \G;
*************************** 1. row ***************************
NOW(): 2018-10-02 20:26:29

DATETIME型はそのまま日時が格納されて返されるが、TIMESTAMP型は UTC として保持して 返すときにtime_zoneに変換される違いがある。 したがって time_zone を変更すると DATETIME のカラムの値は変わらず、TIMESTAMP は変わる。

> CREATE TABLE t (
    dt DATETIME,
    ts TIMESTAMP
);
> INSERT INTO t VALUES (NOW(), NOW());
> select * from t \G;
*************************** 1. row ***************************
dt: 2018-10-02 20:27:13
ts: 2018-10-02 20:27:13

> SET SESSION time_zone = "UTC";
> select NOW() \G;
*************************** 1. row ***************************
NOW(): 2018-10-02 11:27:56

> select * from t \G;
*************************** 1. row ***************************
dt: 2018-10-02 20:27:13
ts: 2018-10-02 11:27:13

各パラメータの関係

go-sql-driver/mysqlで loc と time_zone を付けてDBに接続し、 MySQLの NOW() の値とGoの time.Now() の値を DATETIME型 と TIMESTAMP 型の カラムに格納して出力する。 time.Local は UTC でも MySQLの time_zone の JST でもない US/Alaska (-0800) にしている。

package main

import (
    "database/sql"
    "fmt"
    "math"
    "time"

    _ "github.com/go-sql-driver/mysql"
)
    
const format = "2006-01-02 15:04:05 Z0700"
    
func main() {
    var err error
    if time.Local, err = time.LoadLocation("US/Alaska"); err != nil {
        panic(err)
    }
    now := time.Now()
    fmt.Printf("%15s: %s\n", "time.Now()", now.Format(format))
    fmt.Println("* none")
    run(now, "root:@/hoge?parseTime=true")
    fmt.Println("* loc (mysql's timezone != Local timezone)")
    run(now, "root:@/hoge?parseTime=true&loc=Local")
    fmt.Println("* loc & time_zone")
    run(now, "root:@/hoge?parseTime=true&loc=Local&time_zone=%27US%2FAlaska%27")
}

func run(now time.Time, src string) {
    db, err := sql.Open("database", "mysql", src)
    if err != nil {
        panic(err)
    }
    defer db.Close()

    if _, err := db.Exec("DELETE FROM t"); err != nil {
        panic(err)
    }

    if _, err := db.Exec("INSERT INTO t VALUES (NOW(), NOW())"); err != nil {
        panic(err)
    }

    if _, err := db.Exec("INSERT INTO t VALUES (?, ?)", now, now); err != nil {
        panic(err)
    }

    rows, err := db.Query("SELECT dt, ts FROM t")
    if err != nil {
        panic(err)
    }
    i := 0
    title := []string{"mysql NOW()", "go time.Now()"}
    for rows.Next() {
        var datetime, timestamp time.Time
        if err := rows.Scan(&datetime, &timestamp); err != nil {
            panic(err)
        }
        fmt.Printf("%15s: %s, %s -> %v\n",
            title[i],
            datetime.Format(format),
            timestamp.Format(format),
            math.Abs(float64(datetime.Unix()-now.Unix())) < 10,
        )
        i++
    }
}

実行結果

  • time.Now(): 2018-10-02 03:43:13 -0800
  • set none
    • mysql NOW(): 2018-10-02 20:43:13 Z, 2018-10-02 20:43:13 Z -> false
    • go time.Now(): 2018-10-02 20:43:13 Z, 2018-10-02 20:43:13 Z -> false
  • set loc (mysql’s timezone != Local timezone)
    • mysql NOW(): 2018-10-02 20:43:13 -0800, 2018-10-02 20:43:13 -0800 -> false
    • go time.Now(): 2018-10-02 03:43:14 -0800, 2018-10-02 03:43:14 -0800 -> true
  • set loc & time_zone
    • mysql NOW(): 2018-10-02 03:43:13 -0800, 2018-10-02 03:43:13 -0800 -> true
    • go time.Now(): 2018-10-02 03:43:14 -0800, 2018-10-02 03:43:14 -0800 -> true

loc は time.Time の タイムゾーンで、付けないと UTC になる。 これは MySQL サーバーの time_zone には影響せず MySQL の NOW() の値は JST のままなので 20:43 (+0900) が返り、 これを loc のタイムゾーンの値として扱ってしまう。 time_zone を渡すと SET time_zone され MySQLの NOW() が意図した値を返すようになる。

一方、Go の time.Now() を格納する場合は loc のみでタイムゾーンが正しくハンドリングされて意図した値が入っている。

DATETIME型 と TIMESTAMP型 については 全ての例において値が一致している。 これは上にも書いた通り格納時と取得時で time_zone を変えていないためで、 既に格納されている行の TIMESTAMP型 カラムを time_stamp を渡さないで取得すると意図した値を返さないことがある。

まとめ

loc=Local は渡さないと読み書きともに問題があるので必須。 NOW() や TIMETAMP型 を 扱う際は time_zone も渡さないと意図した値が得られないことがある。 time.Local と MySQL の time_zone が一致している場合はこの問題は起こらないが、それに依存しているなら明示的に渡すべきだと考えている。
